    \begin{tabular}{c|c}
        \toprule
        \textbf{符号} & \textbf{含义} \\
        \midrule
        \(\N\) & 自然数集 \\
        \(\Z\) & 整数集 \\
        \(\Q\) & 有理数集 \\
        \(\R\) & 实数集 \\
        \(\C\) & 复数集 \\
        \(\Measurable(X)\) & \(X\) 上的 Lebesgue 可测集族 \\
        \(\Borel(X)\) & \(X\) 上的 Borel 集族 \\
        \(\Lebout(E)\) & 集合 \(E\) 的外测度 \\
        \(\Leb(E)\) & 集合 \(E\) 的测度 \\
        \(\Char{E}\) & 集合 \(E\) 的特征函数 \\
        \(P\cae\) & 命题 \(P\) 几乎处处成立 \\ 
        \(f_n\aelim f\) & 函数列 \(f_n\) 几乎处处收敛到函数 \(f\) \\
        \(f_n\mlim f\) & 函数列 \(f_n\) 依测度收敛到函数 \(f\) \\
        \(\udif f\) & \(f\) 的上导数 \\
        \(\ldif f\) & \(f\) 的下导数 \\
        \(\Varia_a^b(f)\) & \(f\) 在 \([a,b]\) 上的全变差 \\
        \bottomrule
    \end{tabular}
